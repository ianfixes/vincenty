def to_degrees(mod=false) 
  if mod 
    (self * 180  / Math::PI) % 360
  else 
    self * 180  / Math::PI 
  end
end

def to_radians(mod=false)
  if mod 
    (self * Math::PI / 180) % Math::PI
  else 
    self * Math::PI / 180
  end
end
